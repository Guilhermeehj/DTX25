from transformers import AutoModelForCausalLM, AutoTokenizer import torch import gradio as gr # Carregando modelo e tokenizer model_name = "microsoft/DialoGPT-medium" tokenizer = AutoTokenizer.from_pretrained(model_name) model = AutoModelForCausalLM.from_pretrained(model_name) # Função de resposta def responder(mensagem, historico=[]): nova_entrada = tokenizer.encode(mensagem + tokenizer.eos_token, return_tensors='pt') if historico: historico_ids = torch.cat([tokenizer.encode(m + tokenizer.eos_token, return_tensors='pt') for m in historico], dim=-1) entrada = torch.cat([historico_ids, nova_entrada], dim=-1) else: entrada = nova_entrada resposta_ids = model.generate(entrada, max_length=1000, pad_token_id=tokenizer.eos_token_id) resposta = tokenizer.decode(resposta_ids[:, entrada.shape[-1]:][0], skip_special_tokens=True) historico.append(mensagem) historico.append(resposta) return resposta, historico # Interface Gradio chatbot = gr.Interface( fn=responder, inputs=gr.inputs.Textbox(lines=2, placeholder="Digite sua pergunta..."), outputs="text", title="Chatbot IA Simples", description="Um chatbot simples baseado no modelo DialoGPT.", theme="compact" ) if __name__ == "__main__": chatbot.launch()
